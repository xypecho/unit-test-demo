<template>
  <div class="header">
    <section>
      <label for="ToDoList">ToDoList</label>
      <input type="text" v-model="inputValue" @keyup.enter='addTodolistItem'>
    </section>
  </div>
</template>
 
<script>
export default {
  data () {
    return {
      inputValue: ''
    }
  },
  methods: {
    addTodolistItem () {
      if (this.inputValue) {
        this.$emit('add', this.inputValue)
        this.inputValue = ''
      }
    }
  }
}
</script>
